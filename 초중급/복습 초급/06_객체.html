  • 객체Object - 접근,추가 삭제 접근: Superman.name //'clark' Superman.['age'] //33 추가: Superman.gender = 'male'; Superman['hairColor'] = 'black'; 예) name: 'clark', age : 33, gender : 'male', hairColor : 'black' 삭제: delete Superman.hairColor;
  • 객체Object - 단축 프로퍼티 const name = 'clark'; const age = 33; const Superman ={ name, //name:clark 와 동일하다. age, //age:age gender:'male', }
